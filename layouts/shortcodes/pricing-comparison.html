{{- /* Pricing Comparison Shortcode Displays original price, current price, and savings Usage: {{< pricing-comparison original="99.97" current="29.97" >}} Or with percentage: {{< pricing-comparison original="99.97" current="29.97" discount_percent="70" >}} */ -}} {{- $original := .Get "original" | float -}} {{- $current := .Get "current" | float -}} {{- $currency := .Get "currency" | default "$" -}} {{- $savings := sub $original $current -}} {{- $discountPercent := .Get "discount_percent" -}} {{- if not $discountPercent -}} {{- $discountPercent = printf "%.0f" (mul (div $savings $original) 100) -}} {{- end -}}
Regular Price: {{ $currency }}{{ $original }}
Today's Price: {{ $currency }}{{ $current }}
You Save: {{ $currency }}{{ printf "%.2f" $savings }} ({{ $discountPercent }}% OFF)
